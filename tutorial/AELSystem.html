L-System plant in AEPanels


(def cut-style {:stroke (p-color 255 0 0)})
(def engrave-style {:stroke (p-color 0 0 255)})

(def wide 25.0)
(def high 101.0)
(def screw-y 3.4)
(def screw-x (/ wide 2))
(def cut-rect  (rect 0 0 wide high  cut-style))
(def gap wide)

(defn circ [cx cy r style] (poly cx cy r 30 style))
(def screw-holes 
   (stack
       (circ screw-x screw-y 1.5 cut-style) 
       (circ screw-x (- high screw-y) 1.5 cut-style)
   )
)

;; Define the custom panel

(defn rcos []
   (let [rc (- 1 (/ (rand-int 20) 20))]
      [(rc) (rc)]
   )
)

(defn project-to-panel [x p]
  (->> p reframe
      (translate-to -1 -1)
      (stretch 12 35)
      (translate 5 17)
     
  )
)

(defn around-2 []
   (+ 1.5 (/ (rand-int 10) 10))
)

(defn plant [d s]
  (->>
     (let
       [grow (l-system [["F" s]])]
         (basic-turtle
           [(/ wide 2) (/ high 2)] (around-2) (/ PI -2) (/ PI d)  (grow 3 "F")  {}  engrave-style
         )
     )
   (wobble [0.2 0.2])
  )
)

(defn panel [x] 
    (->>
    (stack
       (rect 0 0 wide high cut-style)
       screw-holes
       (project-to-panel x 
          (plant (rand-nth [ 3.6 5 7 9  11]) 
            (rand-nth ["F[+F]F[-F][F+F]"
                             "F[-F]F[+F][F-F]"])  )
       )
    )
    (translate x 0)
    )
)


(->>
  (apply stack (take 4
    (map panel (iterate #(+ % gap) 0))
  ))
  reframe
  
  (stack (rect -1 -1 2 2 cut-style) )
)
