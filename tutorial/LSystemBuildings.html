(def lsys (l-system [["B" "BCB"] 
                               ["C" "DBB"]]))

(def  b1 
   (stack
       (rect -1 0 2 1 {})
        [(->SShape {}
          [ [-1 0] [0 -0.4] [1 -0] ] 
   )]
   )
)

(def  b2
   (stack
       (rect -1 -0.5 2 1.5 {})
        [(->SShape {}
          [ [-1 -0.5] [0 -1] [1 -0.5] ] 
   )]
   )
)

(def b3
   (rect -1 0.5 2 0.5 {})
)

(def city (lsys 3 "B"))

(defn block [c dist]
   (->> (condp = c
      \B b1
      \C b2
      \D b3
   ) (translate dist 0))
)

(defn p [s]
   (apply stack
     (map block s (range 0 100 2)))
)

(reframe (p city))
